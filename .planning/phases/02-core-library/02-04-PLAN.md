Create dependency audit and verify no std leakage:

1. Create scripts/audit_std_leak.sh:
   ```bash
   #!/bin/bash
   set -e
   TARGET=thumbv7em-none-eabihf

   echo "Checking for std leakage in no_std build..."
   cargo tree --target $TARGET --no-default-features | grep -i "std" && {
       echo "ERROR: std dependency found in no_std build!"
       exit 1
   } || echo "PASS: No std dependencies found"

   echo "Checking for HashMap (requires Random trait)..."
   cargo tree --target $TARGET --no-default-features | grep -i "hashmap" && {
       echo "WARNING: HashMap found (may not work in no_std)"
   } || echo "PASS: No HashMap found"

   echo "All checks passed!"
   ```
